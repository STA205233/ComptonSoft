177  template <typename Func>
178  void doForEachMultiChannelDataInDetectorOrder(const Func& func)
179  {
180  for (std::unique_ptr<VRealDetectorUnit>& detector: detectors_) {
181  const int NumMCD = detector->NumberOfMultiChannelData();
182  for (int i=0; i<NumMCD; i++) {
183  MultiChannelData* mcd = detector->getMultiChannelData(i);
184  DetectorBasedChannelID channel(detector->getID(), i);
185  func(mcd, channel);
186  }
187  }
188  }

190  template <typename Func>
191  void doForEachMultiChannelDataInReadoutOrder(const Func& func)
192  {
193  for (std::unique_ptr<ReadoutModule>& readoutModule: readoutModules_) {
194  const int NumMCD = readoutModule->NumberOfSections();
195  for (int i=0; i<NumMCD; i++) {
196  const int readoutModuleID = readoutModule->ID();
197  const ReadoutBasedChannelID readoutChannel(readoutModuleID, i);
198  MultiChannelData* mcd = this->getMultiChannelData(readoutChannel);
199  func(mcd, readoutChannel);
200  }
201  }
